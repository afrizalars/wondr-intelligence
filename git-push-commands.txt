# After creating the repository on GitHub, run these commands:

# Add the remote repository (replace YOUR_USERNAME with your GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/wondr-intelligence.git

# Or if using SSH:
# git remote add origin git@github.com:YOUR_USERNAME/wondr-intelligence.git

# Push to GitHub
git branch -M main
git push -u origin main

# If you want to install GitHub CLI and create repo from command line:
brew install gh
gh auth login
gh repo create wondr-intelligence --private --source=. --remote=origin --push
